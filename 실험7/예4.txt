	MOVLW	B'10110110'
	MOVWF	COM_B
	CALL	RS232C
	CALL	DELAY		; 안정된 동작을 위하여
	MOVLW	B'10000001'
	MOVWF	COM_B
	CALL	RS232C
	CALL	DELAY		; 안정된 동작을 위하여
	MOVLW	B'10011111'
	MOVWF	COM_B
	CALL	RS232C
	CALL	DELAY		; 안정된 동작을 위하여
	MOVLW	B'10100001'
	MOVWF	COM_B
	CALL	RS232C
	CALL	DELAY		; 안정된 동작을 위하여
BBLP	
	GOTO	BBLP

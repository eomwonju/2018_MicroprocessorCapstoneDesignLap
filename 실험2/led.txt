
	PROCESSOR 16F876A
	INCLUDE <P16F876A.INC>

	ORG	0000
	BSF	STATUS, RP0	; BANK를 1로 변경함
	MOVLW	B'00000001'
	MOVWF	TRISA
	MOVLW	B'00000000'
	MOVWF	TRISC
	MOVLW 	B'00000111'	; PORTA를 DIGITAL I/O로 사용
	MOVWF	ADCON1
	BCF	STATUS, RP0	; BANK를 0으로 변경
	
LOOP
	BTFSS	PORTA, 0
	GOTO	L1
	MOVLW	0FH		; 입력 1일때의 출력
	MOVWF	PORTC
	GOTO	LOOP

L1
	MOVLW	0F0H		; 입력 0일때의 출력
	MOVWF	PORTC
	GOTO	LOOP

	END
